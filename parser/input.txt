CREATE TABLE empleados (
    id INT PRIMARY KEY INDEX BTREE,
    nombre VARCHAR(20) INDEX HASH,
    sueldo FLOAT INDEX AVL,
    activo BOOL INDEX HASH,
    fecha_ingreso DATE INDEX BTREE
);

SELECT id, nombre, salario, fecha_ingreso
FROM empleados
WHERE (salario > 3000 AND departamento = 'ventas')
   OR (activo = TRUE AND (antiguedad > 5 OR fecha_ingreso < '2020-01-01'))
   AND nombre != 'Juan'
   AND NOT eliminado
   AND edad BETWEEN 25 AND 60;
